        def self.valid_date?(y,m,d)
                (y.class == Fixnum && y > 0 && 
                 m.class == Fixnum && (1..12).include?(m) && 
                 d.class == Fixnum && (((1..JDaysInMonth[m-1]).include?(d)) || (d == 30 && m == 12 && leap?(y)   )) 
                ) ? true : false
        end
